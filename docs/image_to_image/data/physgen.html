 39def resize_tensor_to_divisible_by_14(tensor: torch.Tensor) -> torch.Tensor:
 40    """
 41    Resize a tensor so that its height and width are divisible by 14.
 42
 43    This function ensures the spatial dimensions (H, W) of a given tensor 
 44    are compatible with architectures that require sizes divisible by 14 
 45    (e.g., ResNet, ResFCN). It resizes using bilinear interpolation.
 46
 47    Parameter:
 48    - tensor (torch.Tensor): 
 49        Input tensor of shape (C, H, W) or (B, C, H, W).
 50
 51    Returns:
 52    - torch.Tensor: 
 53        Resized tensor with dimensions divisible by 14.
 54
 55    Raises:
 56    - ValueError: 
 57        If the tensor has neither 3 nor 4 dimensions.
 58    """
 59    if tensor.dim() == 3:
 60        c, h, w = tensor.shape
 61        new_h = h - (h % 14)
 62        new_w = w - (w % 14)
 63        return F.interpolate(tensor.unsqueeze(0), size=(new_h, new_w), mode='bilinear', align_corners=False).squeeze(0)
 64    
 65    elif tensor.dim() == 4:
 66        b, c, h, w = tensor.shape
 67        new_h = h - (h % 14)
 68        new_w = w - (w % 14)
 69        return F.interpolate(tensor, size=(new_h, new_w), mode='bilinear', align_corners=False)
 70    
 71    else:
 72        raise ValueError("Tensor must be 3D (C, H, W) or 4D (B, C, H, W)")

 79class PhysGenDataset(Dataset):
 80    """
 81    PyTorch Dataset wrapper for the PhysicsGen dataset.
 82
 83    Loads the PhysGen dataset from Hugging Face and provides configurable
 84    input/output modes for physics-based generative learning tasks.
 85
 86    The dataset contains Open Sound Maps (OSM) and simulated soundmaps
 87    for tasks involving sound propagation modeling.
 88    """
 89    def __init__(self, variation="sound_baseline", mode="train", input_type="osm", output_type="standard", 
 90                 fake_rgb_output=False, make_14_dividable_size=False,
 91                 reflexion_channels=False, reflexion_steps=36, reflexions_as_channels=False):
 92        """
 93        Loads PhysGen Dataset.
 94
 95        Parameter:
 96        - variation (str, default='sound_baseline'): 
 97            Dataset variation to load. Options include:
 98            {'sound_baseline', 'sound_reflection', 'sound_diffraction', 'sound_combined'}.
 99        - mode (str, default='train'): 
100            Dataset split to use. Options: {'train', 'test', 'validation'}.
101        - input_type (str, default='osm'): 
102            Defines the input image type:
103            - 'osm': open sound map input.
104            - 'base_simulation': uses the baseline sound simulation as input.
105        - output_type (str, default='standard'): 
106            Defines the output image type:
107            - 'standard': full soundmap prediction.
108            - 'complex_only': difference from baseline soundmap.
109        - fake_rgb_output (bool, default=False): 
110            If True, replicates single-channel inputs to fake RGB (3-channel).
111        - make_14_dividable_size (bool, default=False): 
112            If True, resizes tensors so that height and width are divisible by 14.
113        - reflexion_channels (bool, default=False): 
114            If ray-traces should add to the input.
115        - reflexion_steps (int, default=36): 
116            Defines how many traces should get created.
117        - reflexions_as_channels (bool, default=False): 
118            If True, every trace gets its own channel, else every trace in one channel.
119        """
120        self.fake_rgb_output = fake_rgb_output
121        self.make_14_dividable_size = make_14_dividable_size
122        self.reflexion_channels = reflexion_channels
123        self.reflexion_steps = reflexion_steps
124        self.reflexions_as_channels = reflexions_as_channels
125
126        self.device = 'cuda' if torch.cuda.is_available() else 'mps' if torch.backends.mps.is_available() else 'cpu'
127        # get data
128        self.dataset = load_dataset("mspitzna/physicsgen", name=variation, trust_remote_code=True)
129        # print("Keys:", self.dataset.keys())
130        self.dataset = self.dataset[mode]
131        
132        self.input_type = input_type
133        self.output_type = output_type
134        if self.input_type == "base_simulation" or self.output_type == "complex_only":
135            self.basesimulation_dataset = load_dataset("mspitzna/physicsgen", name="sound_baseline", trust_remote_code=True)
136            self.basesimulation_dataset = self.basesimulation_dataset[mode]
137
138        self.transform = transforms.Compose([
139            transforms.ToTensor(),  # Converts [0,255] PIL image to [0,1] FloatTensor
140        ])
141        print(f"PhysGen ({variation}) Dataset for {mode} got created")
142
143    def __len__(self):
144        """
145        Returns the number of available samples.
146
147        Returns:
148        - int: 
149            Number of samples in the dataset split.
150        """
151        return len(self.dataset)
152
153    def __getitem__(self, idx):
154        """
155        Retrieve an input-target pair from the dataset.
156
157        This function loads the input image and corresponding target image,
158        applies transformations (resizing, fake RGB, etc.), and returns
159        them as PyTorch tensors.
160
161        Parameter: 
162        - idx (int): 
163            Index of the data sample.
164
165        Returns: 
166        - tuple[torch.Tensor, torch.Tensor]<br>
167            A tuple containing:
168            - input_img : torch.Tensor<br>
169                Input tensor (shape: [C, H, W]).
170            - target_img : torch.Tensor<br>
171                Target tensor (shape: [C, H, W]).
172        """
173        sample = self.dataset[idx]
174        # print(sample)
175        # print(sample.keys())
176        if self.input_type == "base_simulation":
177            input_img = self.basesimulation_dataset[idx]["soundmap"]
178        else:
179            input_img = sample["osm"]  # PIL Image
180        target_img = sample["soundmap"]  # PIL Image
181
182        input_img = self.transform(input_img)
183        target_img = self.transform(target_img)
184
185        # Fix real image size 512x512 > 256x256
186        input_img = F.interpolate(input_img.unsqueeze(0), size=(256, 256), mode='bilinear', align_corners=False)
187        input_img = input_img.squeeze(0)
188        # target_img = target_img.unsqueeze(0)
189
190        # change size
191        if self.make_14_dividable_size:
192            input_img = resize_tensor_to_divisible_by_14(input_img)
193            target_img = resize_tensor_to_divisible_by_14(target_img)
194
195        # add fake rgb
196        if self.fake_rgb_output and input_img.shape[0] == 1:  # shape (1, H, W)
197            input_img = input_img.repeat(3, 1, 1)  # make it (B, 3, H, W)
198
199        if self.output_type == "complex_only":
200            base_simulation_img = self.transform(self.basesimulation_dataset[idx]["soundmap"])
201            # base_simulation_img = resize_tensor_to_divisible_by_14(self.transform(self.basesimulation_dataset[idx]["soundmap"]))
202            # target_img = torch.abs(target_img[0] - base_simulation_img[0])
203            target_img = target_img[0] - base_simulation_img[0]
204            target_img = target_img.unsqueeze(0)
205            target_img *= -1
206
207        # add raytracing
208        if self.reflexion_channels:
209            rays = ips.ray_tracing.trace_beams(rel_position=(0.5, 0.5),	
210                                                img_src=np.squeeze(input_img.cpu().numpy(), axis=0),	
211                                                directions_in_degree=ips.math.get_linear_degree_range(step_size=(self.reflexion_steps/360)*100),	
212                                                wall_values=[0],	
213                                                wall_thickness=0,	
214                                                img_border_also_collide=False,	
215                                                reflexion_order=3,	
216                                                should_scale_rays=True,	
217                                                should_scale_img=False)
218            ray_img = ips.ray_tracing.draw_rays(rays,	
219                                                detail_draw=False,	
220                                                output_format='channels' if self.reflexions_as_channels else 'single_image',	
221                                                img_background=None,	
222                                                ray_value=[50, 100, 255],	
223                                                ray_thickness=1,	
224                                                img_shape=(256, 256),
225                                                should_scale_rays_to_image=True,
226                                                show_only_reflections=True)
227            # (256, 256)
228            # print("CHECKPOINT")
229            # print(ray_img.shape)
230            ray_img = self.transform(ray_img)
231            ray_img = ray_img.float()
232            if ray_img.ndim == 2:
233                ray_img = ray_img.unsqueeze(0)  # (1, H, W)
234
235            # print(ray_img.shape)
236            # print(input_img.shape)
237            # Merging with input image 
238            if ray_img.shape[1:] == input_img.shape[1:]:
239                input_img = torch.cat((input_img, ray_img), dim=0)
240            else:
241                raise ValueError(f"Ray image shape {ray_img.shape} does not match input image shape {input_img.shape}.")
242
243        return input_img, target_img

280def get_image(mode='train', variation="sound_reflection", input_type="osm", output_type="complex_only", shuffle=True, 
281              return_output=False, as_numpy_array=True):
282    """
283    Retrieve one image (input and optionally output) from the PhysGen dataset.
284
285    Provides an easy way to visualize or inspect a single PhysGen sample
286    without manually instantiating a DataLoader.
287
288    Parameter:
289    - mode (str, default='train'): 
290        Dataset split ('train', 'test', 'validation').
291    variation (str, default='sound_reflection'): 
292        Dataset variation.
293    input_type (str, default='osm'): 
294        Defines the input type ('osm' or 'base_simulation').
295    output_type (str, default='complex_only'): 
296        Defines the output type ('standard' or 'complex_only').
297    shuffle (bool, default=True): 
298        Randomly select the sample.
299    return_output (bool, default=False): 
300        If True, returns both input and target tensors.
301    as_numpy_array (bool, default=True): 
302        If True, converts tensors to NumPy arrays for easier visualization.
303
304    Returns: 
305    - numpy.ndarray or list[numpy.ndarray]: 
306        Input image as NumPy array, or a list [input, target] if `return_output` is True.
307    """
308    dataset = PhysGenDataset(mode=mode, variation=variation, input_type=input_type, output_type=output_type)
309    loader = DataLoader(dataset, batch_size=1, shuffle=shuffle, num_workers=1)
310    cur_data = next(iter(loader))
311    input_ = cur_data[0]
312    output_ = cur_data[1]
313
314    if as_numpy_array:
315        input_ = input_.detach().cpu().numpy()
316        output_ = output_.detach().cpu().numpy()
317
318        # remove batch channel
319        input_ = np.squeeze(input_, axis=0)
320        output_ = np.squeeze(output_, axis=0)
321
322        if len(input_.shape) == 3:
323            input_ = np.squeeze(input_, axis=0)
324            output_ = np.squeeze(output_, axis=0)
325
326        input_ = np.transpose(input_, (1, 0))
327        output_ = np.transpose(output_, (1, 0))
328
329
330    result = input_
331    if return_output:
332        result = [input_, output_]
333
334    return result

338def save_dataset(output_real_path, output_osm_path, 
339                 variation, input_type, output_type,
340                 data_mode, 
341                 info_print=False, progress_print=True):
342    """
343    Save PhysGen dataset samples as images to disk.
344
345    This function loads the specified PhysGen dataset, converts input and
346    target tensors to images, and saves them as `.png` files for inspection,
347    debugging, or model-agnostic data use.
348
349    Parameter: 
350    - output_real_path (str): 
351        Directory to save target (real) soundmaps.
352    - output_osm_path (str): 
353        Directory to save input (OSM) maps.
354    - variation (str): 
355        Dataset variation (e.g. 'sound_reflection').
356    - input_type (str): 
357        Input type ('osm' or 'base_simulation').
358    - output_type (str): 
359        Output type ('standard' or 'complex_only').
360    - data_mode (str): 
361        Dataset split ('train', 'test', 'validation').
362    - info_print (bool, default=False): 
363        If True, prints detailed information for each saved sample.
364    - progress_print (bool, default=True): 
365        If True, shows progress updates in the console.
366
367    Raises:
368    - ValueError: 
369        If image data falls outside the valid range [0, 255].
370
371    """
372    # Clearing
373    if os.path.exists(output_osm_path) and os.path.isdir(output_osm_path):
374        shutil.rmtree(output_osm_path)
375        os.makedirs(output_osm_path)
376        print(f"Cleared {output_osm_path}.")
377    else:
378        os.makedirs(output_osm_path)
379        print(f"Created {output_osm_path}.")
380
381    if os.path.exists(output_real_path) and os.path.isdir(output_real_path):
382        shutil.rmtree(output_real_path)
383        os.makedirs(output_real_path)
384        print(f"Cleared {output_real_path}.")
385    else:
386        os.makedirs(output_real_path)
387        print(f"Created {output_real_path}.")
388    
389    # Load Dataset
390    dataset = PhysGenDataset(mode=data_mode, variation=variation, input_type=input_type, output_type=output_type)
391    data_len = len(dataset)
392    dataloader = DataLoader(dataset, batch_size=1, shuffle=False, num_workers=4)
393
394    # Save Dataset
395    for i, data in enumerate(dataloader):
396        if progress_print:
397            prime.get_progress_bar(total=data_len, progress=i+1, 
398                                   should_clear=True, left_bar_char='|', right_bar_char='|', 
399                                   progress_char='#', empty_char=' ', 
400                                   front_message='Physgen Data Loading', back_message='', size=15)
401
402        input_img, target_img, idx = data
403        idx = idx[0].item() if isinstance(idx, torch.Tensor) else idx
404
405        if info_print:
406            print(f"Prediction shape [osm]: {input_img.shape}")
407            print(f"Prediction shape [target]: {target_img.shape}")
408
409            print(f"OSM Info:\n    -> shape: {input_img.shape}\n    -> min: {input_img.min()}, max: {input_img.max()}")
410
411        real_img = target_img.squeeze(0).cpu().squeeze(0).detach().numpy()
412        if not (0 <= real_img.min() <= 255 and 0 <= real_img.max() <=255):
413            raise ValueError(f"Real target has values out of 0-256 range => min:{real_img.min()}, max:{real_img.max()}")
414        if info_print:
415            print( f"\nReal target has values out of 0-256 range => min:{real_img.min()}, max:{real_img.max()}")
416        if real_img.max() <= 1.0:
417            real_img *= 255
418        if info_print:
419            print( f"Real target has values out of 0-256 range => min:{real_img.min()}, max:{real_img.max()}")
420        real_img = real_img.astype(np.uint8)
421        if info_print:
422            print( f"Real target has values out of 0-256 range => min:{real_img.min()}, max:{real_img.max()}")
423
424        if len(input_img.shape) == 4:
425            osm_img = input_img[0, 0].cpu().detach().numpy()
426        else:
427            osm_img = input_img[0].cpu().detach().numpy()
428        if not (0 <= osm_img.min() <= 255 and 0 <= osm_img.max() <=255):
429            raise ValueError(f"Real target has values out of 0-256 range => min:{osm_img.min()}, max:{osm_img.max()}")
430        if osm_img.max() <= 1.0:
431            osm_img *= 255
432        osm_img = osm_img.astype(np.uint8)
433
434        if info_print:
435            print(f"OSM Info:\n    -> shape: {osm_img.shape}\n    -> min: {osm_img.min()}, max: {osm_img.max()}")
436
437        # Save Results
438        file_name = f"physgen_{idx}.png"
439
440        # save pred image
441        # save_img = os.path.join(output_pred_path, file_name)
442        # cv2.imwrite(save_img, pred_img)
443        # print(f"    -> saved pred at {save_img}")
444
445        # save real image
446        save_img = os.path.join(output_real_path, "target_"+file_name)
447        cv2.imwrite(save_img, real_img)
448        if info_print:
449            print(f"    -> saved real at {save_img}")
450
451        # save osm image
452        save_img = os.path.join(output_osm_path, "input_"+file_name)
453        cv2.imwrite(save_img, osm_img)
454        if info_print:
455            print(f"    -> saved osm at {save_img}")
456    print(f"\nSuccessfull saved {data_len} datapoints into {os.path.abspath(output_real_path)} & {os.path.abspath(output_osm_path)}")
